NAME
  =tar - how to decompress some files

DESCRIPTION
  =tar is my personal manual to decompress some files using tar command
  
  - .tgz, .tar.gz, gz
    
    $ tar -vzxf filename.tar.gz

  - .tgz, .tar.gz , .gz
 	
    $ tar -vzxf filename.tar.gz

  - .tar.bz2 
 	
    $ tar -vxjpf filename.tar.bz2

  - .tar 
 	
    $ tar xf filename.tar

  - .rar
 	
    $ rar x filename.rar

  -	.zip
 	
    $ unzip filename.zip

  - .7-zip
	
    $ 7z x -y filename.7z

  - .tar.xz
 	  
    $ tar -xf arquivo.tar
